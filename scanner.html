function startAutoDetect(){

  detectionActive=true;let stable=0;

  const check=()=>{

    if(!detectionActive)return;

    const video=document.getElementById("video"),w=video.videoWidth,h=video.videoHeight;

    if(!w){requestAnimationFrame(check);return;}

    const canvas=document.createElement("canvas");canvas.width=Math.min(w,320);canvas.height=Math.min(h,240);

    const ctx=canvas.getContext("2d");ctx.drawImage(video,0,0,canvas.width,canvas.height);

    const data=ctx.getImageData(0,0,canvas.width,canvas.height).data;

    let brightness=0,edgeCount=0;

    for(let i=0;i<data.length;i+=16)brightness+=(data[i]+data[i+1]+data[i+2])/3;

    brightness/=(data.length/16);

    for(let y=1;y<canvas.height-1;y+=3)for(let x=1;x<canvas.width-1;x+=3){const idx=(y*canvas.width+x)*4;if(Math.abs(data[idx]-data[(y*canvas.width+x+1)*4])+Math.abs(data[idx]-data[((y+1)*canvas.width+x)*4])>30)edgeCount++;}

    const er=edgeCount/(canvas.width*canvas.height/9),isDoc=brightness>60&&brightness<230&&er>0.04&&er<0.45;

}

  else if(cropPointer.mode==="tl"){const r=cropPointer.rect.x+cropPointer.rect.width,b=cropPointer.rect.y+cropPointer.rect.height;next.x=clamp(cropPointer.rect.x+dx,cropBounds.left,r-min);next.y=clamp(cropPointer.rect.y+dy,cropBounds.top,b-min);next.width=r-next.x;next.height=b-next.y;}

  else if(cropPointer.mode==="tr"){const b=cropPointer.rect.y+cropPointer.rect.height;next.y=clamp(cropPointer.rect.y+dy,cropBounds.top,b-min);next.width=clamp(cropPointer.rect.width+dx,min,cropBounds.left+cropBounds.width-cropPointer.rect.x);next.height=b-next.y;}

  else if(cropPointer.mode==="bl"){const r=cropPointer.rect.x+cropPointer.rect.width;next.x=clamp(cropPointer.rect.x+dx,cropBounds.left,r-min);next.width=r-next.x;next.height=clamp(cropPointer.rect.height+dy,min,cropBounds.top+cropBounds.height-cropPointer.rect.y);}

  else if(cropPointer.mode==="br"){next.width=clamp(cropPointer.rect.width+dx,min,cropBounds.left+cropBounds.width-cropPointer.rect.x);next.height=clamp(cropPointer.rect.height+dy,min,cropBounds.top+cropBounds.height-cropPointer.rect.y);}

  cropRect=next;renderCropBox();

}

async function getProcessedBlob(){

  if(!capturedDataUrl||!cropRect||!cropBounds)return capturedBlob;

  const img=new Image();img.src=capturedDataUrl;await img.decode();

  const sx=Math.round((cropRect.x-cropBounds.left)*(img.naturalWidth/cropBounds.width)),sy=Math.round((cropRect.y-cropBounds.top)*(img.naturalHeight/cropBounds.height));

  const sw=Math.round(cropRect.width*(img.naturalWidth/cropBounds.width)),sh=Math.round(cropRect.height*(img.naturalHeight/cropBounds.height));

  const canvas=document.createElement("canvas");canvas.width=sw;canvas.height=sh;

  canvas.getContext("2d").drawImage(img,sx,sy,sw,sh,0,0,sw,sh);

  enhanceImage(canvas.getContext("2d"),sw,sh);

  let fw=sw,fh=sh;if(Math.max(fw,fh)>1600){const sc=1600/Math.max(fw,fh);fw=Math.round(fw*sc);fh=Math.round(fh*sc);const c2=document.createElement("canvas");c2.width=fw;c2.height=fh;c2.getContext("2d").drawImage(canvas,0,0,fw,fh);canvas.width=fw;canvas.height=fh;canvas.getContext("2d").drawImage(c2,0,0);}

  return await new Promise((resolve,reject)=>{const tryQ=(q)=>{canvas.toBlob(b=>{if(!b){reject(new Error("blob"));return;}if(b.size<=490*1024||q<=0.4)resolve(b);else tryQ(Math.round((q-0.08)*100)/100);},"image/jpeg",q);};tryQ(0.85);});

}
