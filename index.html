      contentBase64: base64

    };


    const res = await fetch(STORE_URL, {

      method:'POST',

      headers:{'Content-Type':'application/json'},

      body: JSON.stringify(payload)

    });

    if (!res.ok) throw new Error('HTTP ' + res.status);


    setSlotStatus(slot, 'sent');

    // حفظ في sessionStorage

    const sd = JSON.parse(sessionStorage.getItem('pai_done_slots_session') || '[]');

    if (!sd.includes(slot)) { sd.push(slot); sessionStorage.setItem('pai_done_slots_session', JSON.stringify(sd)); }

    hideWaitingBanner();

    addHistory(doc?.ar || `Slot ${slot}`, true);


  } catch(e) {

    setSlotStatus(slot, 'error');

    hideWaitingBanner();

    addHistory(`Slot ${slot}`, false);

  }


  // لا reset - تبقى "تم" حتى إعادة التحميل

}

function setSlotStatus(slot, status) {

  slotStatus[slot] = status;

  const card  = document.getElementById('card_' + slot);

  const badge = document.getElementById('badge_' + slot);

  if (!card || !badge) return;


  card.classList.remove('card-uploading');


  const states = {

    idle:      { text:'جاهز',       cls:'bg-surface-container-high text-on-surface-variant' },

    uploading: { text:'جاري الرفع…', cls:'bg-primary-fixed text-primary' },

    sent:      { text:'✓ تم', cls:'bg-green-100 text-green-700' },

    error:     { text:'✗ فشل',       cls:'bg-error-container text-error' },

  };


  const s = states[status] || states.idle;

  badge.textContent = s.text;

  badge.className = `text-[10px] font-bold px-2 py-0.5 rounded-full shrink-0 ${s.cls}`;

  if (status === 'uploading') card.classList.add('card-uploading');

}

async function sendPassport() {

  if (!passportFile || sending) return;

  sending = true;

  const btn = document.getElementById('passportSendBtn');

  btn.disabled = true;

  btn.innerHTML = '<span class="inline-block animate-spin mr-2">⏳</span> جاري الإرسال...';


  try {

    const blob = await processImage(passportFile);

    const fd = new FormData();

    fd.append('data', blob, passportFile.name.replace(/\.[^.]+$/, '') + '-enhanced.jpg');

    fd.append('_source', 'sovereign-mobile');


    const res = await fetch(EXTRACT_URL, { method:'POST', body:fd });

    const text = await res.text();

    let json = null;

    try { json = JSON.parse(text); } catch { const m=text.match(/\{[\s\S]*\}/); if(m) try{json=JSON.parse(m[0]);}catch{} }

    if (!res.ok) throw new Error('HTTP ' + res.status);

    if (Array.isArray(json)) json = json[0]||{};

    if (json?.data && typeof json.data==='object') json = json.data;


    showPassportResult(json);

    addHistory(passportFile.name, true);


  } catch(e) {

    document.getElementById('passportResult').classList.remove('hidden');

    document.getElementById('passportFields').innerHTML = `<p class="text-sm text-error">${e.message}</p>`;

    addHistory(passportFile.name, false);

  }


  sending = false;

  btn.disabled = false;

  btn.textContent = '🚀 إرسال واستخراج';

}

    }

  // استرجع المستندات المكتملة

  const sessionDone = JSON.parse(sessionStorage.getItem('pai_done_slots_session') || '[]');

  sessionDone.forEach(slot => setSlotStatus(slot, 'sent'));


  // رجوع من السكانر بـ ?done=

  const urlDone = Number(new URLSearchParams(location.search).get('done') || 0);

  if (urlDone) {

    if (!sessionDone.includes(urlDone)) {

      sessionDone.push(urlDone);

      sessionStorage.setItem('pai_done_slots_session', JSON.stringify(sessionDone));

    }

    setSlotStatus(urlDone, 'sent');

    history.replaceState({}, '', location.pathname);

  }

  const slot = sessionStorage.getItem('pai_pending_slot');

  const img  = sessionStorage.getItem('pai_scanned_image');

  const name = sessionStorage.getItem('pai_scanned_name') || 'scanned.jpg';

  if (slot && img) {

    sessionStorage.removeItem('pai_pending_slot');

    sessionStorage.removeItem('pai_scanned_image');

    sessionStorage.removeItem('pai_scanned_name');

    fetch(img).then(r=>r.blob()).then(blob => {

      const file = new File([blob], name, {type:'image/jpeg'});

      if (slot === 'passport') selectPassportFile(file);

      else sendDocumentSlot(Number(slot), file);

    });

  }

});


/* ─── معالجة الصورة ─── */

async function processImage(f) {

  return new Promise((res, rej) => {

    const reader = new FileReader();

    reader.onload = () => {

      const img = new Image();

      img.onload = () => {

        const s = Math.min(1, 1600 / Math.max(img.width, img.height));

        const w = Math.round(img.width * s), h = Math.round(img.height * s);

        const c = document.createElement('canvas'); c.width = w; c.height = h;

        const ctx = c.getContext('2d'); ctx.drawImage(img, 0, 0, w, h);

        const d = ctx.getImageData(0,0,w,h);

        for(let i=0;i<d.data.length;i+=4){

          d.data[i]=Math.min(255,Math.max(0,((d.data[i]-128)*1.1)+130));

          d.data[i+1]=Math.min(255,Math.max(0,((d.data[i+1]-128)*1.1)+130));

          d.data[i+2]=Math.min(255,Math.max(0,((d.data[i+2]-128)*1.1)+130));

        }

        ctx.putImageData(d,0,0);

        const MAX_BYTES = 490 * 1024;

        const tryQuality = (q) => {

          c.toBlob(b => {

            if (!b) { rej(new Error('blob failed')); return; }

            if (b.size <= MAX_BYTES || q <= 0.4) { res(b); }

            else { tryQuality(Math.round((q - 0.08) * 100) / 100); }

          }, 'image/jpeg', q);

        };

        tryQuality(0.85);

      };

      img.onerror = () => rej(new Error('image load failed'));

      img.src = reader.result;

    };

    reader.onerror = () => rej(new Error('file read failed'));

    reader.readAsDataURL(f);

  });

}


async function blobToBase64(blob) {

  return new Promise((res,rej) => {

    const r=new FileReader();

    r.onload=()=>res(String(r.result||'').split('base64,')[1]||'');

    r.onerror=()=>rej(new Error('read'));

    r.readAsDataURL(blob);

  });

}


/* ─── History ─── */

function addHistory(name, ok) {

  history.unshift({name, ok, ts:Date.now()});

  if(history.length>20) history=history.slice(0,20);

  localStorage.setItem('pai_hist', JSON.stringify(history));

  renderHistory();

}

function clearHistory() {

  history=[];

  localStorage.removeItem('pai_hist');

  renderHistory();

}

function renderHistory() {

  const el = document.getElementById('historyList');

  if(!history.length){ el.innerHTML='<p class="text-sm text-on-surface-variant text-center py-4">لا يوجد سجل</p>'; return; }

  el.innerHTML = history.map(h => {

    const d = new Date(h.ts).toLocaleString('ar',{hour:'2-digit',minute:'2-digit',day:'2-digit',month:'2-digit'});

    return `<div class="bg-surface-container-lowest rounded-xl px-4 py-2.5 flex items-center gap-3 editorial-shadow">

      <div class="flex-1">

        <p class="text-xs font-bold text-on-surface truncate">${(h.name||'').slice(0,30)}</p>

        <p class="text-[11px] text-on-surface-variant">${d}</p>

      </div>

      <span class="text-[11px] font-bold px-2 py-0.5 rounded-full ${h.ok?'bg-green-100 text-green-700':'bg-error-container text-error'}">${h.ok?'✓ نجح':'✗ فشل'}</span>

    </div>`;

  }).join('');

}
